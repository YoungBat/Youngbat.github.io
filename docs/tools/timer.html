This is a timer demo.


Code:

        var second;
second=0;
var clock;
function resetTimer()
{
    clearInterval(clock);
    second=0;
    document.getElementById('timeValue').value=second+' sec';
}
function startTimer()
{
    clearInterval(clock);
    clock=setInterval(timer,1000);
}
function stopTimer() {
    clearInterval(clock);
    document.getElementById('timeValue').value=second+' sec';
}
function timer(){
    second++;
    document.getElementById('timeValue').value=second+' sec';
}
